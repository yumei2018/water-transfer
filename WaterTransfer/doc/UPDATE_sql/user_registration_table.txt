CREATE TABLE USER_REGISTRATION(
  USER_ID NUMERIC(10,0) PRIMARY KEY NOT NULL,
  FIRST_NAME VARCHAR2(16 BYTE),
  MIDDLE_NAME VARCHAR2(16 BYTE),
  LAST_NAME VARCHAR2(16 BYTE),
  EMAIL VARCHAR2(32 BYTE),
  USERNAME VARCHAR2(16 BYTE),
  PASSWORD VARCHAR2(32 BYTE),
  ACTIVE NUMBER(1,0) DEFAULT 1,
  CREATED_DATE TIMESTAMP(6) default SYSTIMESTAMP,
  UPDATED_DATE TIMESTAMP(6) default SYSTIMESTAMP,
  WT_CONTACT_ID NUMBER(16,0)
);

CREATE SEQUENCE USER_REGISTER_PKSEQ;


create or replace
TRIGGER USER_REGISTER_PKTRIG 
BEFORE INSERT ON USER_REGISTRATION
FOR EACH ROW 
BEGIN 
  IF :NEW.USER_ID IS NULL THEN
    SELECT USER_REGISTER_PKSEQ.NEXTVAL INTO :NEW.USER_ID FROM DUAL;
  END IF;  
  SELECT SYSTIMESTAMP INTO :NEW.CREATED_DATE FROM DUAL;
  SELECT SYSTIMESTAMP INTO :NEW.UPDATED_DATE FROM DUAL;
END USER_REGISTER_PKTRIG;
